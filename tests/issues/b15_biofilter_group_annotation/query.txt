SELECT
    m_bg.label AS gene_label,
    d_g.label AS group_label,
    m_c.label AS source_label,
    (COALESCE(m_bg.biopolymer_id,'')||'_'||COALESCE(d_g.group_id,'')||'_'||COALESCE(m_c.source_id,'')) AS _rowid
FROM 
    `db`.`group` AS d_g,
    `db`.`group_biopolymer` AS d_gb,
    main`.`gene` AS m_bg,
    `main`.`source` AS m_c
WHERE 
    (d_gb.specificity >= 100 OR d_gb.specificity >= 100)  AND
    d_g.group_id = d_gb.group_id  AND
    d_g.source_id = m_c.source_id  AND
    d_gb.biopolymer_id != 0  AND
    d_bg.source_id != 0 AND
    d_g.source_id != 0 AND
    d_gb.group_id = d_g.group_id AND
    m_bg.biopolymer_id = d_gb.biopolymer_id AND
    m_c.source_id = d_g.source_id
